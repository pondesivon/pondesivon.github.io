Sayı Değişken Türlerinin Alt Ve Üst Sınırlarını Öğren - Java

Sayı ile ilgili değişken türlerinin alt ve üst değerlerini yazdıran java kodudur.

Kod

public class JavaD1
{
	public static void main(String[] args)
	{
		System.out.println("Byte Sinirlari");
		System.out.println(Byte.MIN_VALUE);
		System.out.println(Byte.MAX_VALUE);
		
		System.out.println("Short Sinirlari");
		System.out.println(Short.MIN_VALUE);
		System.out.println(Short.MAX_VALUE);

		System.out.println("Integer Sinirlari");
		System.out.println(Integer.MIN_VALUE);
		System.out.println(Integer.MAX_VALUE);		
		
		System.out.println("Long Sinirlari");
		System.out.println(Long.MIN_VALUE);
		System.out.println(Long.MAX_VALUE);
		
		System.out.println("Float Sinirlari");
		System.out.println(Float.MIN_VALUE);
		System.out.println(Float.MAX_VALUE);
		
		System.out.println("Double Sinirlari");
		System.out.println(Double.MIN_VALUE);
		System.out.println(Double.MAX_VALUE);
	}
}
