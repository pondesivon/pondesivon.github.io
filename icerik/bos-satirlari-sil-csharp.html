public static string BosSatirlariSil(string metin)
{
	string[] dizi = { };

	if (metin.Contains('\n'))
	{
		dizi = metin.Split('\n');
	}
	else if (metin.Contains('\r'))
	{
		dizi = metin.Split('\r');
	}
	else if (metin.Contains(Convert.ToChar(Environment.NewLine)))
	{
		dizi = metin.Split(Convert.ToChar(Environment.NewLine));
	}

	StringBuilder sbd = new StringBuilder();

	for (int i = 0; i < dizi.Length; i++)
	{
		if (dizi[i] != String.Empty)
		{
			sbd.AppendLine(NRTS_Trim(dizi[i]));
		}
	}

	return sbd.ToString();
}
