İki Sayı İle 4 İşlem Yap - Microsoft Excel

Verilen iki sayı ile toplama, çıkarma, çarpma ve bölme işlemlerinin yapıldığı ve sonucun mesaj olarak gösterildiği makro kodudur.

Kod

Sub IkiSayiIle4IslemYap()
    Dim sayi1 As Double
    Dim sayi2 As Double
    Dim toplama, cikarma, carpma, bolme As Double
    
    sayi1 = 10
    sayi2 = 4
    
    toplama = sayi1 + sayi2
    cikarma = sayi1 - sayi2
    carpma = sayi1 * sayi2
    bolme = sayi1 / sayi2
    
    MsgBox sayi1 & vbCrLf & sayi2 & String(2, vbCrLf) & _
           "Toplama: " & toplama & vbCrLf & _
                     "Çıkarma: " & cikarma & vbCrLf & _
                     "Çarpma: " & carpma & vbCrLf & _
                     "Bölme: " & bolme
End Sub
