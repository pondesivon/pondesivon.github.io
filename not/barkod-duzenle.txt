Sub BarkodDuzenle()
    Dim i As Long
    
    For i = 4 To [E1048576].End(xlUp).Row
        If InStr(Range("E" & i), "-") > 0 Then
            Range("E" & i) = Replace(Range("E" & i), "-", "")
        End If
    Next
End Sub

Sub BarkodDuzenleAlan(alan As Range)
    alan = Replace(alan, "-", "")
    alan.NumberFormat = "0"
End Sub
